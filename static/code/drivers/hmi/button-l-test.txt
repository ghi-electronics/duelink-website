##### Button L Test #####

fn Test()
    println("Starting test...")
    Wait(1)
    dlmode(2)
    Wait(10)
    cmd("sel(1)")
    while 1        
        if (dread(BtnPin(), 1))
            cmd("Clear(0)")
            cmd("TextS(\"Button up\",1,0,0,2,2)")
            cmd("Show()") 
            println("Button up")
        else
            cmd("Clear(0)")
            cmd("TextS(\"Button pressed\",1,0,0,2,2)")
            cmd("Show()")
            println("Button pressed")
        end
        Wait(200)
        
    wend
fend
