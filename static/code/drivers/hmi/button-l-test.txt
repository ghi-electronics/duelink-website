##### Button S Test #####

fn Test()
    println("Starting test...")
    Wait(1)
    dlmode(2)
    Wait(10)
    cmd("sel(1)")
    while 1        
        if (dread(BtnPin(), 1))
            cmd("Clear(0)")
            cmd("TextS(\n"Button pressed\n",1,0,0,1,1)")
            cmd("Show()")
        else
            cmd("Clear(0)")
            cmd("TextS(\n"Button up\n",1,0,0,1,1)")
            cmd("Show()")
        end
        Wait(100)
        
    wend
fend
