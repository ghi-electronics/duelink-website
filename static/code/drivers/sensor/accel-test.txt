##### Led RGB3 Test #####

dim b1[50]
fn Test() 
    println("Starting test...")
    Wait(1)
    dlmode(2)
    Wait(10)
    cmd("sel(1)")
    
    while 1                 
        x = GetX()
        y = GetY()
        z = GetZ()
        
        cmd("clear(0)")
        
        sprintf(b1, "Texts(\"x=%d\",1,0,0,2,2)", x)        
        cmd(b1)
        
        sprintf(b1, "Texts(\"y=%d\",1,0,20,2,2)", y)        
        cmd(b1)
        
        sprintf(b1, "Texts(\"z=%d\",1,0,40,2,2)", z)        
        cmd(b1)
        
        cmd("show()")
        Wait(1000)       
    wend
fend
