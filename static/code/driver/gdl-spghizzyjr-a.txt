fn DVer()

fend 

fn Init()
    BtnEn(BtnPin(), 1) # Btn LDR
fend

fn BtnPin()
    return 20
fend

fn BtnLdr()
    if BtnDown(BtnPin())
    return 1
fend

fn Eye(l,r)
    DWrite(4,(l >> 0) & 0xFF)
    DWrite(5,(l >> 16) & 0xFF)
    DWrite(6,(l >> 8) & 0xFF)
 
    DWrite(7,(r >> 0) & 0xFF)
    DWrite(8,(r >> 16) & 0xFF)
    DWrite(9,(r >> 8) & 0xFF)
fend

Init()
