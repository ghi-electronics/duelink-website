##### Keypad 4x4 #####

fn Rset()
    cmd("reset(0)") # reset my client
    reset(0) # reset myself
fend

fn Test()  
    println("Starting test...")
    Wait(1)
    dlmode(2,0)
    Wait(10)
    cmd("sel(1)")
        
    
    while 1 

        if IsKeyChange()
            c = chr(RdKey())
            PrintLn("key pressed: ", c)
            cmd(fmt("Texts(\"key pressed:", c, "\",1,0,0,1,2)"))
        end
        
        wait(10)
    wend
fend
