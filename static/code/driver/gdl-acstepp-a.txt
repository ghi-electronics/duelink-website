fn DVer()
    return 1.0
fend

fn Set(s,d,t,a,b)
    if d != 1
        d = 0
    end
    
    dwrite(b,d)

    for i = 0 to s
        dwrite(a,1)
        waitus(t)
        dwrite(a,0)
        waitus(t)
    next
fend

fn SetX(s,d,t)
    Set(s,d,t,3,4)
fend

fn SetY(s,d,t)
    Set(s,d,t,5,6)
fend

fn SetZ(s,d,t)
    Set(s,d,t,9,8)
fend
