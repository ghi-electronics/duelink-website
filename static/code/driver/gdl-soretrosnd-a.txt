##### Retro Sound #####

fn DVer()
    return 1.0
fend

fn GetPin()
    return 1
fend
 
fn SetLVol(v)
    if v >=100
        dwrite(3,1)
    else
        freq(3,48000, 0, v/100)
    end
fend

fn SetRVol(v)
    if v >=100
        dwrite(4,1)
    else
        freq(4,48000, 0, v/100)
    end
fend

fn Init() 
    SetLVol(100)
    SetRVol(100)
fend

Init()
