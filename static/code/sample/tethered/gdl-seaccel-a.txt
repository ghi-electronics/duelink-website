# In this sample:
# Reading X,Y,Z values every second

DLMode(2,0) # Switch Downlink to host mode
Wait(100) # This delay is more than enough for 50x modules
cmd("sel(1)")#Select First Module

fn GetX()
    return cmd("GetX()")
fend

fn GetY()
    return cmd("GetY()")
fend

fn GetZ()
    return cmd("GetZ()")
fend

while 1
    PrintLn("x = ", GetX())
    PrintLn("y = ", GetY())
    PrintLn("z = ", GetZ())
    Wait(1000)
wend
