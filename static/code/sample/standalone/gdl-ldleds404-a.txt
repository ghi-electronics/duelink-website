# In this sample:
# Display the current minute and second on the LED (timer version).
# Time start from 00:00

m = 0
s = 0
while 1 
    Seg7(0, floor(s % 10))
    Seg7(1, floor(s / 10))
    Seg7(2, floor(m % 10))
    Seg7(3, floor(m / 10))
    
    colon(1)
    Show()
    wait(500)
    
    colon(0)
    Show()
    wait(500)
    
    s = s + 1
    
    if (s = 60)
        s = 0
        m = m + 1
        
        if (m = 60)
            m = 0
        end
    end
wend
