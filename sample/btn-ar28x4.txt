while 1
    if BtnA()
        PrintLn("A pressed")
        SetClr(0, Rnd(6), 100)
    end
    if BtnB()
        PrintLn("B pressed")
        SetClr(1, Rnd(6), 100)
    end
    if BtnC()
        PrintLn("C pressed")
        SetClr(2, Rnd(6), 100)
    end
    if BtnD()
        PrintLn("D pressed")
        SetClr(3, Rnd(6), 100)
    end

    Wait(30)
wend
