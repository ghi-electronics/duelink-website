_a = 0
_b = 0
BtnEn(BtnPin(),0)

while 1
    _x = JoystickX()
    _y = JoystickY()
    
    if(_a < _x - 1 || _a > _x + 1)
        PrintLn("X = ", _x)
        _a = _x
    end
    
    if(_b < _y - 1 || _b > _y + 1)
        PrintLn("Y = ", _y)
        _b = _y
    end
    
    if(BtnDown(BtnPin()))
        PrintLn("Joystick Pressed")
    end

    Wait(10)
wend
