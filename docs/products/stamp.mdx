##### Stamp Driver #####

fn DVer()
    return 1.0
fend
 
fn Init()
    BtnEn(BtnPin(), 0)
fend

fn BtnPin()
    return 20
fend

fn BtnLdr()
    if BtnDown(BtnPin())
    return 1
fend

Init()
