fn CalCrc(b4, s)
    h = 0xff
    l = 0xff
    i = 0
    
    for x = 0 to s
        i = l ^ b4[x]
        l = h ^ b1[i]
        h = b2[i]
    next
    
    return (h << 8) | l
fend

_r = 5 # re pin
_d = 6 # de pin

fn Init() 
    SerCfg(9600, 128)
    DWrite(_r, 0) # This pin can be low all the time
fend


fn GetDevId() 
    Dim b5[8]
    Dim b4[7] = [0,0,0,0,0,0,0]
    
    b5[0] = 0x00
    b5[1] = 0x03
    b5[2] = 0x40
    b5[3] = 0
    b5[4] = 0
    b5[5] = 1
    b5[6] = 0x90
    b5[7] = 0x1B
    
    SerDisc() # Clear all received data in buffer
    
    DWrite(_d, 1)
    SerWrs(b5) # send new command
    DWrite(_d, 0)

    # Wait for response  around 5ms
    while SerB2R() = 0
        Wait(5)
    wend
    
    #Wait for full response
    while SerB2R() < 7
        Wait(5)
    wend
    
    # read and calculate device id
    SerRds(b4, 1000)
     
    # Clear all response, we just want to set channel by writting
    SerDisc()
    
    c = CalCrc(b4, 5)
    
    if ((c & 0xFF) = b4[5] && ((c>>8) & 0xFF) = b4[6]) 
         # device id 16 bit
        d = (b4[3] << 8) | (b4[4] << 0)
        PrintLn("good crc, dev = ", d)
        
        return 1
    else
        PrintLn("bad crc")
        
        return 0
    end
    
    
fend

fn SetCh(n, v) 
    Dim b5[8]
    Dim b4[6]
    
    b5[0] = 0x01
    b5[1] = 0x06
    b5[2] = n >> 8
    b5[3] = n & 0xFF
    b5[4] = v >> 8
    b5[5] = v & 0xFF
    
    # copy b5 to b4, 6 element for calculation crc
    Memcpy(b4, 0, b5, 0, 6)
    c =  CalCrc(b4,6)
    
    b5[6] = c & 0xFF
    b5[7] = c >> 8
    
    SerDisc() # Clear all received data in buffer
    
    DWrite(_d, 1)
    SerWrs(b5) # send new command
    DWrite(_d, 0)
     
    # Wait for response
    while SerB2R() = 0
        Wait(5)
    wend
    
    #Wait for full response
    while SerB2R() < 8
        Wait(5)
    wend
    
    # Clear all response, we just want to set channel by writting
    SerDisc() 
fend

fend
