| `BtnA()` | Returns 1 if button A is pressed. |
| `BtnB()` | Returns 1 if button B is pressed. |
| `IsTouch(p)` | Returns 1 if `p` pad (0, 1, or 2) is touched. |
| `Buzzer(f,d)` | Plays buzzer to `f` frequency(50 to 15K) for `d` duration in milliseconds. |
| `Light()` | Returns voltage from light sensor. |

**CincoBit Driver**

```batch
##### CincoBit Driver #####

fn DVer()
    return 1.0
fend

fn Init()
    BtnEn(BtnPin('A'), 1) # Btn A
    BtnEn(BtnPin('B'), 1) # Btn B
    GfxCfg(4, a1, 5,5,1) # 4: 5x5 type, 0: direct
fend

fn BtnPin(a)
    if a = 'A'
        return 20
    end
    if a = 'B'
        return 23
    end
    while 1
        PrintLn("ERROR!!")
        Wait(200)
    wend
fend

Dim a1[11] = {0, 11,19,18,21,22,7,5,0,8,6} # 0: flag, next 5 for data pins, next 5 for scan pins

fn IsColor()
    return 0
fend

fn GetW()
    return 5
fend

fn GetH()
    return 5
fend

fn BtnA()
    if BtnDown(BtnPin('A')) 
        PrintLn("A pressed")
        return 1
    else
        return 0
    end
fend

fn BtnB()
    if BtnDown(BtnPin('B'))
        PrintLn("B pressed")
        return 1
    else
        return 0
    end
fend

fn Buzzer(f,d)
    Beep(4, f, d)
fend

fn IsTouch(p)
    return Touch(p+1,50,1,100) > 200
fend

fn Light()
    return Trunc(VRead(9)/(ReadVCC() )*100)
fend

Init()

#########################
```

### Python

```python
from DUELink.DUELinkController import DUELinkController
import time
availablePort = DUELinkController.GetConnectionPort()
duelink = DUELinkController(availablePort)
x = 0
while True:
    duelink.Graphics.TextT("DUELink", 1, x, 0)
    duelink.Graphics.Show()
    x -= 1
    if x < -40:
        x = 6
    time.sleep(0.1)
    duelink.Graphics.Clear(0)
```

### Script

```batch
while 1
    TextT("DUELink", 1, _x, 0)
    Show()
    _x= _x-1

    if _x < -40
        _x = 6
    end
    
    if BtnA()
        PrintLn("Button A Pressed")
        Buzzer(300,500)
    end
    
    if BtnB()
        PrintLn("Button B Pressed")
        Buzzer(700,500)
    end
    
    if Light() < 1
        Buzzer(1000,500)
    end
    
    if (IsTouch(0) || IsTouch(1) || IsTouch(2))
        PrintLn("Pad Touched")
        Buzzer(1500,100)
    end
    
    Wait(100)
    Clear(0) # need to clear for next character
wend
```
